启用 Https
获取 Nginx 证书
开启 443 端口（Https 服务）
1
# HTTPS server server { listen 443 ssl; # 1.1版本后这样写 server_name www.laijunjie.com; #填写绑定证书的域名 ssl_certificate "C:/dir/1_www.laijunjie.com_bundle.crt"; # 证书的位置，绝对路径 ssl_certificate_key "C:/dir/2_www.laijunjie.com.key"; # 绝对路径，同上 ssl_session_timeout 5m; ssl_protocols TLSv1 TLSv1.1 TLSv1.2; #按照这个协议配置 ssl_ciphers ECDHE-RSA-AES128-GCM-SHA256:HIGH:!aNULL:!MD5:!RC4:!DHE;#按照这个套件配置 ssl_prefer_server_ciphers on; location / { root ../public; #站点目录，绝对路径 index index.html index.htm; } }
